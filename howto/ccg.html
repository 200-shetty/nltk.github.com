>>> from nltk.ccg import chart, lexicon

Construct a lexicon:

>>> lex = lexicon.parseLexicon('''
...     :- S, NP, N, VP
...
...     Det :: NP/N
...     Pro :: NP
...     Modal :: S\\NP/VP
...
...     TV :: VP/NP
...     DTV :: TV/NP
...
...     the => Det
...
...     that => Det
...     that => NP
...
...     I => Pro
...     you => Pro
...     we => Pro
...
...     chef => N
...     cake => N
...     children => N
...     dough => N
...
...     will => Modal
...     should => Modal
...     might => Modal
...     must => Modal
...
...     and => var\\.,var/.,var
...
...     to => VP[to]/VP
...
...     without => (VP\\VP)/VP[ing]
...
...     be => TV
...     cook => TV
...     eat => TV
...

Tests handling of conjunctions. Note that while the two derivations are different, they are semantically equivalent.
