[docs]    def lemmatize(self, word, pos=NOUN):
        lemmas = _wordnet._morphy(word, pos)
        if not lemmas:
            return word
        lemmas.sort(key=len)
        return lemmas[0]


    def __repr__(self):
        return '<WordNetLemmatizer>'
